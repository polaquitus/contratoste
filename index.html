Plan mínimo de intervención · Tarifarios con BD

No más localStorage como solución final. La persistencia debe seguir el mismo camino del contrato.
Objetivo validado: importar listas (1..3 tablas), guardarlas en contract.tarifarios, pisar por período + nombre, y generar nuevas listas ajustadas desde un período base manteniendo la misma estructura de tablas y líneas.

1) Qué sí tocar

  • standardizeExcelPriceList y parsePriceListExcelFile para soportar TARIFARIO FINAL con hasta 3 tablas.
  • importPriceListsFromFiles para hacer upsert en contract.tarifarios (pisar mismo período + mismo nombre).
  • generateSelectedPriceLists para replicar la misma estructura del período base y aplicar el mismo porcentaje a todas las celdas numéricas tarifarias.

No tocar

  • getSelectedPeriodsSummaryRows: ya devuelve {ym, refYm, pct} a partir de la selección y la base.
  • openConditionsModal, evaluateConditions, detectFirstCompliance y renders de input type="date/month".
  • guardarEnm salvo que después quieras unificar también el flujo por enmienda.

2) Dónde está la clave

La lógica correcta ya aparece en guardarEnm() cuando el tipo es ACTUALIZACION_TARIFAS: toma las tablas base (bTars), recorre todas las columnas y aplica el ajuste usando isNumericCol, luego persiste el contrato con sbUpsertItem('contratos', cc) si SB_OK.

La diferencia es que generateSelectedPriceLists() hoy sólo busca una columna precio/valor unitario. El fix propuesto la alinea con la misma regla funcional que ya usa guardarEnm().

Persistencia BD:
  • Primero intentar save() si existe.
  • Si no, usar el mismo patrón que guardarEnm(): if (SB_OK) sbUpsertItem('contratos', cc).
  • El fallback a localStorage queda sólo como contingencia técnica, no como comportamiento objetivo.

3) Patch JS listo para incrustar

Pegá este bloque al final del HTML estable, después de las definiciones existentes, o usalo como referencia exacta para reemplazar las funciones.

/* Patch mínimo BD - Tarifarios */
(function(){
  'use strict';

  function normText(s){
    return String(s||'').toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g,'').trim();
  }
  function cleanText(s){ return String(s==null?'':s).replace(/\s+/g,' ').trim(); }
  function num(v){
    if(v==null || v==='') return null;
    if(typeof v==='number') return Number.isFinite(v)?v:null;
    if(typeof normalizeImportedPriceValue==='function'){
      var n = normalizeImportedPriceValue(v);
      return n===''?null:n;
    }
    var s = String(v).trim().replace(/\./g,'').replace(',', '.');
    var n2 = Number(s);
    return Number.isFinite(n2)?n2:null;
  }
  function safePeriod(p){ return String(p||'').slice(0,7); }
  function detectPeriod(blobText, cc, forcedPeriod){
    if(forcedPeriod) return safePeriod(forcedPeriod);
    if(typeof detectPeriodFromText==='function'){
      try{
        var p = detectPeriodFromText(blobText);
        if(p) return safePeriod(p);
      } catch(e){}
    }
    return safePeriod((cc&&cc.btar)||(cc&&cc.fechaIni)||'');
  }
  function rowHasTitle(r,title){
    return (r||[]).some(function(c){ return normText(c).indexOf(normText(title))>=0; });
  }
  function looksLikeTarifarioFinal(sheetName, rows){
    if(normText(sheetName).indexOf('tarifario final')>=0) return true;
    return (rows||[]).slice(0,30).some(function(r){
      return rowHasTitle(r,'SERVICIO BÁSICO MENSUAL') || rowHasTitle(r,'SERVICIO BASICO MENSUAL') ||
             rowHasTitle(r,'EQUIPOS EVENTUALES CON OPERADOR') ||
             rowHasTitle(r,'SERVICIO DE PROVISIÓN DE CALCÁREO') || rowHasTitle(r,'SERVICIO DE PROVISION DE CALCAREO');
    });
  }
  function parseTarifarioFinal(json, period, fileName){
    var rows = Array.isArray(json)?json:[];
    function findStart(title){ return rows.findIndex(function(r){ return rowHasTitle(r, title); }); }
    var iBas = findStart('SERVICIO BÁSICO MENSUAL');
    var iEvt = findStart('EQUIPOS EVENTUALES CON OPERADOR');
    var iCal = findStart('SERVICIO DE PROVISIÓN DE CALCÁREO');

    function buildSection(startIdx, endIdx, mode){
      if(startIdx < 0) return null;
      var slice = rows.slice(startIdx, endIdx > startIdx ? endIdx : rows.length);
      var out = [];
      slice.forEach(function(r){
        var item = r && r[0];
        var desc = cleanText(r && r[1]);
        if(!desc) return;
        if(mode==='basico'){
          var price = num(r && r[2]);
          if(num(item)==null || price==null) return;
          out.push([String(item).trim(), desc, 'MES', price]);
        }
        if(mode==='eventuales'){
          var vals = [num(r&&r[2]), num(r&&r[3]), num(r&&r[4]), num(r&&r[5])];
          if(num(item)==null || vals.every(function(v){ return v==null; })) return;
          out.push([String(item).trim(), desc, vals[0], vals[1], vals[2], vals[3]]);
        }
        if(mode==='calcareo'){
          var p2 = num(r&&r[2]);
          var p3 = num(r&&r[3]);
          var price2 = p3!=null ? p3 : p2;
          if(num(item)==null || price2==null) return;
          var unit = p3!=null ? cleanText(r&&r[2]) : '';
          out.push([String(item).trim(), desc, unit, price2]);
        }
      });
      return out.length ? out : null;
    }

    var basRows = buildSection(iBas, iEvt > 0 ? iEvt : iCal, 'basico');
    var evtRows = buildSection(iEvt, iCal, 'eventuales');
    var calRows = buildSection(iCal, rows.length, 'calcareo');
    var tables = [];
    if(basRows) tables.push({ name:'Servicio Basico Mensual', cols:['N Item','Descripcion','Unidad','Precio'], rows:basRows, period:period, source:'EXCEL_TARIFARIO_FINAL', sourceFileName:fileName, importedAt:new Date().toISOString(), editable:true });
    if(evtRows) tables.push({ name:'Equipos Eventuales con Operador', cols:['N Item','Descripcion','0-15 Dias Diurno','0-15 Dias Nocturno','Mayor 15 Dias Diurno','Mayor 15 Dias Nocturno'], rows:evtRows, period:period, source:'EXCEL_TARIFARIO_FINAL', sourceFileName:fileName, importedAt:new Date().toISOString(), editable:true });
    if(calRows) tables.push({ name:'Servicio de Provision de Calcareo', cols:['N Item','Descripcion','Unidad','Precio'], rows:calRows, period:period, source:'EXCEL_TARIFARIO_FINAL', sourceFileName:fileName, importedAt:new Date().toISOString(), editable:true });
    return tables;
  }

  window.standardizeExcelPriceList = function(sheetName, json, cc, fileName, forcedPeriod){
    if(!Array.isArray(json) || json.length < 1) return null;
    var headers = (json[0]||[]).map(function(v){ return cleanText(v); });
    var h = headers.map(normText);
    var idxBySyn = function(syns){ return h.findIndex(function(x){ return syns.some(function(s){ return x.indexOf(s)>=0; }); }); };
    var itemIdx = idxBySyn(['item','codigo','cod','n item','n° item','numero item','posicion']);
    var descIdx = idxBySyn(['descripcion','detalle','concepto','texto breve','short text','servicio']);
    var unitIdx = idxBySyn(['unidad','uom','um','unit']);
    var priceIdx = idxBySyn(['precio','valor unitario','precio unitario','tarifa','rate','importe','valor']);
    if(descIdx < 0) descIdx = headers.length > 1 ? 1 : 0;
    if(itemIdx < 0) itemIdx = 0;
    if(unitIdx < 0) unitIdx = Math.min(2, headers.length - 1);
    if(priceIdx < 0) priceIdx = Math.min(3, headers.length - 1);
    var rows = json.slice(1).filter(function(r){ return Array.isArray(r) && r.some(function(v){ return String(v||'').trim()!==''; }); }).map(function(r){
      return [cleanText(r[itemIdx]), cleanText(r[descIdx]), cleanText(r[unitIdx]), num(r[priceIdx])];
    }).filter(function(r){ return r.some(function(v){ return String(v??'').trim()!==''; }); });
    if(!rows.length) return null;
    var blobText = [sheetName].concat(json.slice(0,8).flat()).join(' ');
    return {
      name: cleanText(sheetName || 'Lista Excel'),
      cols: ['Item','Descripción','Unidad','Precio'],
      rows: rows,
      period: detectPeriod(blobText, cc, forcedPeriod),
      source: (typeof slugSource==='function' ? slugSource(fileName,'EXCEL') : 'EXCEL'),
      sourceFileName: fileName,
      importedAt: new Date().toISOString(),
      editable: true
    };
  };

  window.parsePriceListExcelFile = async function(file, cc, forcedPeriod){
    if(typeof XLSX==='undefined') throw new Error('No se cargó XLSX');
    var buf = await file.arrayBuffer();
    var wb = XLSX.read(new Uint8Array(buf), {type:'array'});
    var out = [];
    wb.SheetNames.forEach(function(sn){
      var ws = wb.Sheets[sn];
      var json = XLSX.utils.sheet_to_json(ws, {header:1, defval:''});
      if(looksLikeTarifarioFinal(sn, json)){
        var period = detectPeriod([sn].concat(json.slice(0,30).flat()).join(' '), cc, forcedPeriod);
        out.push.apply(out, parseTarifarioFinal(json, period, file.name));
        return;
      }
      var table = window.standardizeExcelPriceList(sn, json, cc, file.name, forcedPeriod);
      if(table) out.push(table);
    });
    return out;
  };

  function upsertImportedTables(contract, imported){
    if(!contract.tarifarios) contract.tarifarios = [];
    imported.forEach(function(newTar){
      contract.tarifarios = (contract.tarifarios||[]).filter(function(oldTar){
        var samePeriod = String(oldTar.period||'') === String(newTar.period||'');
        var sameName = normText(oldTar.name) === normText(newTar.name);
        return !(samePeriod && sameName);
      });
      contract.tarifarios.push(newTar);
    });
  }

  window.importPriceListsFromFiles = async function(files, forcedPeriod){
    if(!files || !files.length) return;
    var cc = (window.DB||[]).find(function(x){ return String(x.id)===String(window.detId); });
    if(!cc){ if(typeof toast==='function') toast('No hay contrato seleccionado','er'); return; }
    var imported = [];
    var errors = [];
    for(const file of Array.from(files)){
      var ext = (file.name.split('.').pop()||'').toLowerCase();
      try{
        if(['xls','xlsx'].includes(ext)){
          var parsed = await window.parsePriceListExcelFile(file, cc, forcedPeriod || null);
          if(!parsed.length) throw new Error('No se detectaron tablas útiles en el Excel');
          imported.push.apply(imported, parsed);
          continue;
        }
        throw new Error('Formato no soportado en esta etapa. Solo Excel.');
      }catch(err){
        errors.push(file.name+': '+(err.message||'Error inesperado'));
      }
    }
    if(!imported.length){ if(typeof toast==='function') toast(errors[0] || 'No se importaron listas','er'); return; }
    upsertImportedTables(cc, imported);
    cc.updatedAt = new Date().toISOString();
    if(typeof save==='function') await Promise.resolve(save());
    else if(window.SB_OK && typeof sbUpsertItem==='function') await sbUpsertItem('contratos', cc);
    else localStorage.setItem('cta_v7', JSON.stringify(window.DB||[]));
    try{ if(typeof renderTarifario==='function') renderTarifario(); }catch(e){}
    if(typeof toast==='function') toast(imported.length+' tabla(s) importada(s) y guardada(s)','ok');
  };

  function detectAdjustableColumns(cols, rows){
    cols = Array.isArray(cols)?cols:[];
    rows = Array.isArray(rows)?rows:[];
    var byName = [];
    cols.forEach(function(c, idx){
      var n = normText(c);
      if(/precio|valor unitario|tarifa|diurno|nocturno|importe|rate/.test(n)) byName.push(idx);
    });
    if(byName.length) return Array.from(new Set(byName));
    var found = [];
    cols.forEach(function(c, idx){
      if(idx < 2) return;
      var seen=0, numericCount=0;
      rows.slice(0,10).forEach(function(r){
        if(!Array.isArray(r)) return;
        var v = r[idx];
        if(v==null || String(v).trim()==='') return;
        seen++;
        if(num(v)!=null) numericCount++;
      });
      if(seen>0 && numericCount===seen) found.push(idx);
    });
    return found;
  }

  window.generateSelectedPriceLists = function(cid){
    var contract = (window.DB||[]).find(function(c){ return String(c.id)===String(cid); });
    if(!contract){ if(typeof toast==='function') toast('Contrato no encontrado','er'); return; }
    var rows = (typeof getSelectedPeriodsSummaryRows==='function' ? getSelectedPeriodsSummaryRows(cid) : []);
    if(!rows || !rows.length){ if(typeof toast==='function') toast('Seleccioná al menos un período','er'); return; }
    var all = (contract.tarifarios||[]).slice();
    var created = 0, updated = 0, skipped = 0;
    rows.forEach(function(sel){
      var refYm = sel.refYm;
      var targetYm = sel.ym;
      var pct = sel.pct;
      if(pct==null || !isFinite(Number(pct))) return;
      var factor = 1 + (Number(pct)/100);
      var baseTables = all.filter(function(t){ return String(t.period||'')===String(refYm); });
      if(!baseTables.length){ skipped++; return; }
      baseTables.forEach(function(refTar){
        if(!refTar || !Array.isArray(refTar.rows) || !refTar.rows.length) return;
        var cols = (refTar.cols||[]).slice();
        var adjCols = detectAdjustableColumns(cols, refTar.rows||[]);
        var newRows = (refTar.rows||[]).map(function(r){
          var rr = (r||[]).slice();
          adjCols.forEach(function(ci){
            var oldVal = num(rr[ci]);
            if(oldVal!=null) rr[ci] = Math.round(oldVal * factor * 100) / 100;
          });
          return rr;
        });
        var newName = String(refTar.name||'Tarifario') + ' · Ajuste ' + targetYm;
        var existing = all.find(function(t){
          return String(t.period||'')===String(targetYm) && normText(t.name)===normText(newName) && String(t.source||'')==='POLI_SELECT';
        });
        if(existing){
          existing.rows = newRows;
          existing.basePeriod = refYm;
          existing.pctApplied = Number(pct);
          existing.updatedAt = new Date().toISOString();
          updated++;
        } else {
          all.push({
            name: newName,
            cols: cols,
            rows: newRows,
            period: targetYm,
            basePeriod: refYm,
            pctApplied: Number(pct),
            source: 'POLI_SELECT',
            sourceTableName: refTar.name || '',
            sourceFileName: refTar.sourceFileName || '',
            importedAt: new Date().toISOString(),
            editable: true
          });
          created++;
        }
      });
    });
    contract.tarifarios = all;
    contract.updatedAt = new Date().toISOString();
    if(typeof save==='function') Promise.resolve(save()).then(function(){
      try{ if(typeof renderTarifario==='function') renderTarifario(); }catch(e){}
      try{ if(typeof renderSelectedPeriodsSummary==='function') setTimeout(function(){ renderSelectedPeriodsSummary(cid); }, 50); }catch(e){}
      if(typeof toast==='function') toast(created+' tabla(s) creada(s) · '+updated+' actualizada(s)'+(skipped?' · '+skipped+' período(s) sin base':''),'ok');
    });
    else if(window.SB_OK && typeof sbUpsertItem==='function') sbUpsertItem('contratos', contract).then(function(){ if(typeof toast==='function') toast('Tarifarios actualizados','ok'); });
    else { localStorage.setItem('cta_v7', JSON.stringify(window.DB||[])); if(typeof toast==='function') toast('Tarifarios actualizados (local)','warn'); }
  };
})();
